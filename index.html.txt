    const API_BASE = "https://TON-BACKEND.onrender.com";
    let appConfig = null;

    function showStatus(message, type = "bad", html = false) {
      const box = document.getElementById("status");
      box.className = `status show ${type}`;
      if (html) {
        box.innerHTML = message;
      } else {
        box.textContent = message;
      }
    }

    function clearStatus() {
      const box = document.getElementById("status");
      box.className = "status";
      box.textContent = "";
      box.innerHTML = "";
    }

    function getCustomerData() {
      return {
        name: document.getElementById("name").value.trim(),
        email: document.getElementById("email").value.trim(),
        birthDate: document.getElementById("birthDate").value.trim(),
        birthTime: document.getElementById("birthTime").value.trim(),
        birthPlace: document.getElementById("birthPlace").value.trim(),
        productType: document.getElementById("productType").value,
        notes: document.getElementById("notes").value.trim()
      };
    }

    function validateCustomerData(data) {
      if (!data.name) return "Le nom est requis.";
      if (!data.email) return "L’email est requis.";
      return null;
    }

    async function loadConfig() {
      const res = await fetch(`${API_BASE}/api/config`);
      const data = await res.json();

      if (!res.ok) {
        throw new Error(data.error || "Impossible de charger la configuration.");
      }

      appConfig = data;
      document.getElementById("productName").textContent = data.productName;
      document.getElementById("productPrice").textContent = `${data.price} ${data.currency}`;
    }

    function loadPayPalSdk(clientId, currency) {
      return new Promise((resolve, reject) => {
        if (window.paypal) {
          resolve();
          return;
        }

        const script = document.createElement("script");
        script.src = `https://www.paypal.com/sdk/js?client-id=${encodeURIComponent(clientId)}&currency=${encodeURIComponent(currency)}`;
        script.onload = () => resolve();
        script.onerror = () => reject(new Error("Chargement du SDK PayPal impossible."));
        document.body.appendChild(script);
      });
    }

    async function renderPayPalButtons() {
      await window.paypal.Buttons({
        style: {
          layout: "vertical",
          shape: "pill",
          label: "paypal"
        },

        createOrder: async function () {
          clearStatus();

          const customerData = getCustomerData();
          const validationError = validateCustomerData(customerData);

          if (validationError) {
            showStatus(validationError, "bad");
            throw new Error(validationError);
          }

          const response = await fetch(`${API_BASE}/api/create-order`, {
            method: "POST",
            headers: {
              "Content-Type": "application/json"
            },
            body: JSON.stringify(customerData)
          });

          const result = await response.json();

          if (!response.ok) {
            const errorMsg = result.error || "Erreur de création de commande.";
            showStatus(errorMsg, "bad");
            throw new Error(errorMsg);
          }

          return result.id;
        },

        onApprove: async function (data) {
          clearStatus();

          const response = await fetch(`${API_BASE}/api/capture-order`, {
            method: "POST",
            headers: {
              "Content-Type": "application/json"
            },
            body: JSON.stringify({
              orderID: data.orderID
            })
          });

          const result = await response.json();

          if (!response.ok) {
            showStatus(result.error || "Erreur lors de la capture du paiement.", "bad");
            return;
          }

          const downloadUrl = `${API_BASE}${result.downloadUrl}`;
          showStatus(
            `Paiement validé.<br><a class="download-link" href="${downloadUrl}" target="_blank" rel="noopener noreferrer">Télécharger votre PDF</a>`,
            "ok",
            true
          );
        },

        onError: function (err) {
          console.error(err);
          showStatus("Erreur PayPal. Vérifie les clés, le mode sandbox/live et l’URL du backend.", "bad");
        }
      }).render("#paypal-button-container");
    }

    async function init() {
      try {
        await loadConfig();
        await loadPayPalSdk(appConfig.paypalClientId, appConfig.currency);
        await renderPayPalButtons();
      } catch (error) {
        console.error(error);
        showStatus(error.message || "Erreur de démarrage du module de paiement.", "bad");
      }
    }

    init();
